public class Solution {
    public static String replace(final String s) {
        return s.replace("a","!")
		.replace("e","!")
		.replace("i","!")
		.replace("o","!")
		.replace("u","!")
		.replace("A","!")
		.replace("E","!")
		.replace("I","!")
		.replace("O","!")
		.replace("U","!");
    }
}
